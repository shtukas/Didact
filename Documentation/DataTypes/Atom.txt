Atoms are content definitions

{
    "uuid"     : String
    "mikuType" : "Atom"
    "unixtime" : Float
    "type"     : "description-only"
    "payload"  : nil
}

{
    "uuid"     : String
    "mikuType" : "Atom"
    "unixtime" : Float
    "type"     : "text"
    "payload"  : nhash # nhash of the blob of text
}

{
    "uuid"     : String
    "mikuType" : "Atom"
    "unixtime" : Float
    "type"     : "url"
    "payload"  : url
}

{
    "uuid"      : String
    "mikuType"  : "Atom"
    "unixtime"  : Float
    "type"      : "aion-point"
    "rootnhash" : nhash # nhash of the aion root
}

{
    "uuid"     : String
    "mikuType" : "Atom"
    "unixtime" : Float
    "type"     : "unique-string"
    "payload"  : uniqueString
}

Unique strings can be anything, but when possible, it's recommended to use "Nx01-<substring>", note that in this case "Nx01" is indeed part of the string.
