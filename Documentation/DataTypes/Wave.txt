
Wave:
    "uuid"             : uuid
    "mikuType"         : "Wave"
    "unixtime"         : unixtime
    "datetime"         : DateTime Iso 8601 UTC Zulu
    "description"      : description

    "nx46"             : "field1", JSON.parsed : Nx46 # Wave pattern
    "lastDoneDateTime" : "field4"              : DateTime Iso 8601 UTC Zulu
    "onlyOnDays"       : "field6", JSON.parsed : null or Array[week days names lowercase english] 
                                                 Indicates on which days the wave can show up
    "field11"          : CoreData reference string
    "priority"         : "ns:high" | "ns:medium" | "ns:low"

Nx46 {
    "type"  : Nx46Type
    "value" : Nx46Value
}

Nx46Type                        Nx46Value

"sticky"                        Integer
"every-n-hours"                 Float
"every-n-days"                  Float
"every-this-day-of-the-week"    'sunday' | 'monday' | 'tuesday' | 'wednesday' | 'thursday' | 'friday' | 'saturday'
"every-this-day-of-the-month"   String, length 2, Day Number
