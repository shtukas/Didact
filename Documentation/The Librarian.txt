
### PrimaryStructure.v1

{
    "mikuType"      : "PrimaryStructure.v1"
    "items"         : nhash to a JSON encoded "PrimaryStructure.v1:Items"
}

### PrimaryStructure.v1:NetworkEdges

{
    "mikuType" : "PrimaryStructure.v1:NetworkEdges"
    "edges"    : Array[NxGraphEdge1]
}

NxGraphEdge1 {
    "mikuType" : "NxGraphEdge1"
    "unixtime" : Float
    "uuid1"    : String
    "uuid2"    : String
    "type"     : "bidirectional" | "arrow" | "none"
}

### PrimaryStructure.v1:Items

{
    "mikuType" : "PrimaryStructure.v1:Items",
    "mapping"  : Map[objectuuid, nhash to NxItemSphere1]
}

NxItemSphere1 {
    "mikuType"   : "NxItemSphere1"
    "item"       : Item or null
    "events"     : Array[AttributeUpdate.v2]
}

{
    "mikuType"   => "AttributeUpdate.v2"
    "objectuuid" => objectuuid
    "eventuuid"  => String
    "eventTime"  => Float
    "attname"    => string
    "attvalue"   => value
}
